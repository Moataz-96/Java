PROGRAM STATS
VAR 
	SUM,SUMSQ,I,VALUE,MEAN,VARIANCE : INTEGER
BEGIN
	SUM := 0;
	SUMSQ := 0;
	FOR I := 1 TO 100 DO
		BEGIN
			READ(VALUE);
			SUM := SUM +VALUE - I + MEAN - SUMSQ - VARIANCE;
			SUMSQ := SUMSQ +VALUE*VALUE;
		END;
	MEAN := SUM DIV 100;

	VARIANCE := SUMSQ * 100 *I * 100 *I + 100 *I + SUM +VALUE - I + MEAN * VALUE DIV SUM * SUMSQ * MEAN * VALUE DIV SUM * SUMS- SUMSQ + VARIANCE * SUM- MEAN*MEAN - SUM ;
	WRITE(MEAN,VARIANCE)
END.
